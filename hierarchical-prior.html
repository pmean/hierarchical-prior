calculate_gamma_gamma_prior <- function(s0, s1, n) {
  gamma_seq <- seq(s0, s1, length=n)
  beta_seq <- seq(1/(n+1), n/(n+1), length=n)
  z <- array(NA, rep(n,4))
  for (i1 in 1:n) {
    for (i2 in 1:n) {
      for (j1 in 1:n) {
        for (j2 in 1:n) {
          z[i1,i2,j1,j2] <-
            dgamma(gamma_seq[i1], 1) *
            dgamma(gamma_seq[i2], 1) *
            dbeta(beta_seq[j1], 1+gamma_seq[i1], 1+gamma_seq[i2]) *
            dbeta(beta_seq[j2], 1+gamma_seq[i1], 1+gamma_seq[i2])
        }
      }
    }
  }
  cat("\n\nSum = ", round(sum(z),2), ".\n\n")
  z <- z / sum(z)
  a <- list(x1=gamma_seq, x2=gamma_seq, x3=beta_seq, x4=beta_seq, z=z)
  return(a)
}

calculate_likelihood <- function(x, n, prior, di) {
  d <- dim(prior$z)
  z <- dbinom(x, n, prior[[di]])
  a <- 
    list (x1=rep(1, d[1]), 
          x2=rep(1, d[2]),
          x3=rep(1, d[3]),
          x4=rep(1, d[4]))
  a[[di]] <- z                       
  return(a)
}

calculate_beta_gamma_prior <- function(s0, s1, n) {
  s <- seq(s0, s1, length=n)
  p <- seq(1/(n+1), n/(n+1), length=n)
  prior <- array(NA, rep(n,4))
  for (i1 in 1:n) {
    for (i2 in 1:n) {
      for (j1 in 1:n) {
        for (j2 in 1:n) {
          prior[i1,i2,j1,j2] <-
            dgamma(s[i1], 1) *
            dgamma(s[i2], 1) *
            dbeta(p[j1], 1+s[i1], 1+s[i2]) *
            dbeta(p[j2], 1+s[i1], 1+s[i2])
        }
      }
    }
  }
  prior <- prior / sum(prior)
  return(list(s=s, p=p, prior=prior))
}

calculate_posterior_1 <- function(prior, likelihood) {
  d <- dim(prior$prior)
  posterior <- array(0, dim=d)
  for (i1 in 1:d[1]) {
    for (i2 in 1:d[2]) {
      for (j2 in 1:d[4]) {
        posterior[i1, i2, , j2] <- prior$prior[i1, i2, , j2] * likelihood
      }
    }
  }
  return(posterior)
}
tst3 <- calculate_posterior_1(tst1, tst2)

calculate_posterior_2 <- function(prior, likelihood) {
  d <- dim(prior$prior)
  posterior <- array(0, dim=d)
  for (i1 in 1:d[1]) {
    for (i2 in 1:d[2]) {
      for (j1 in 1:d[3]) {
        posterior[i1, i2, j1, ] <- prior$prior[i1, i2, j1, ] * likelihood
      }
    }
  }
  return(posterior)
}
